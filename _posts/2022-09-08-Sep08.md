# atcoder
自作型の比較関数を設定する。

pytohnについては普通に二次元配列にすれば良い。

'''c
#include <iostream>
#include <cmath>
#include <algorithm>
#include <queue>
using namespace std;

struct Movie{
  int l, r;
};

bool operator<(const Movie &a1, const Movie &a2){
  if (a1.r < a2.r) return true;
  if (a1.r > a2.r) return false;
  if (a1.l < a2.l) return true;
  return false;
}
int N;
Movie A[300009];
int current_time = 0;
int ans = 0;

int main(){
  cin >> N;
  for(int i=1; i<=N; i++){
    cin >> A[i].l >> A[i].r;
  }
  sort(A+1, A+N+1);
  
  for(int i=1; i<=N; i++){
	if(current_time <= A[i].l){
      ans += 1;
      current_time = A[i].r;
    }
  }
  cout << ans << endl;
  
  return 0;
}

'''
